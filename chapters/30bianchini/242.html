IV.2. The orchestra

I will describe only some of the instruments of the orchestra, because they are many, and because often an instrument differs from another just for the name of the soundfile its soundin uses (when Csound will allow us to pass strings from score to orchestra?).

Instrument 1 can generate a very narrow band of filtered noise, and uses butterbp band pass filter. As this unit generator cannot rescale output amplitude, it is important to experiment, to avoid obtaining too low or too high amplitudes (remember the beginning of Bruno Maderna’s “Notturno”?).

Instrument 2 is a better version of “Macchu Picchu” instrument 4, and generates bandpass filtered pulses.

Instrument 3 is a simple FM instrument, but allows to put in the score amplitude envelope attack and release times, and FM index envelope attack and release times. But in it you can see a mechanism of default attack and releases.

Instrument 4 is similar to “Howl” instrument 2, but includes a default calculation of amplitude envelope attack and release times, and a further p-field, p9, which allow “narrowing” or “widening” the span of the frequencies involved, in a manner similar to the well known McAdams’s “spectral fusion”. It uses only 7 oscili, instead of the 10 of “Howl”.

;dannati.orc
sr = 44100
kr = 4410
ksmps = 10
nchnls = 2
;**************** INSTRUMENTS ****************************
;1 = RANDOM NOISE BW .5 HZ
;2 = FILTERED PULSE
;3 = FM
;4 = FLOU
;5 = WHITE NOISE (BASSDRUM LIKE)
;6 = WANDERING TUBA
;7 = LOSCIL
;8 = RANDOM NOISE (SELECTABLE BANDWIDTH)
;9 = GROUPS OF FILTERED PULSES
;10 = SAMPLED BASSDRUM
;11 = RANDOM INDEX FM
;12 = PLUCK (KARPLUS & STRONG)
;13 = FM (FOR “DEJO MEAS. VIEJOS LIBROS...”)
;15 = TRUMPET ADSYN 1 (tr1b.het)
;16 = TRUMPET ADSYN 2 (tr1c.het)
;20 = RING MODULATED VOICE (FOR “L’AGGRESSIONE...")
;...
;56 = Al principio feci la luce...
;57 = L'Europa ha assunto...
;96 = VARIABLE DELAY(USES ga2)
;97 = DELAY1(USESga2)
;98 = DELAY2(USESga3)
;99 = REVERB (USES ga1)
;**********************************************************
ga1 init 0
ga2 init 0
ga3 init 0
;-------------------------------------------------------
; INSTRUMENT 1 RANDOM NOISE
;-------------------------------------------------------
instr 1
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=rand init seed
;p8=rev send to i99
ifrq = cpspch(p5)
iamp = ampdb(p4)
kfrq randi ifrq*.1,1,p7 ;random varying center frequency...
kfrq = ifrq+kfrq ;...summed to desired frequency
kenv linseg 0,p3/8,1,p3*3/4,1,p3/8,0 ;amplitude anvelope
a1 rand iamp ;generate noise...
a2 butterbp a1,ifrq+kfrq,.5 ;...and filter it
a2 = a2 * kenv*20 ;multiplicate amplitude by 20, to
;compensate filter energy losses
ga1 = ga1 + a2 * p8
outs a2 *(1-p6),a2*p6
endin
;-------------------------------------------------------
; INSTRUMENT 2 FILTERED PULSE
;-------------------------------------------------------
instr 2
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo
;p7=bandwidth (cps) (if p7==0 then bw=ifrq/24)
;p8=rev. send
ifrq = cpspch(p5)
iamp = ampdb(p4)
if p7==0 igoto fixbw ;default bandwidth?
ibw = p7 ;no!
igoto ok
fixbw:
ibw = ifrq/24 ;yes, default bandwidth
ok:
a1 linseg iamp,.0025,0,p3-.0025,0 ;generate a pulse,...
a2 butterbp a1,ifrq,ibw ;...filter it...
a3 = a2*10 ;and multiply amplitude
;by 10 to compensate
;filter energy losses
ga1 = ga1+a3*p8
outs a3*(1-p6),a3*p6
endin
;-------------------------------------------------------
; INSTRUMENT 3 FM
;-------------------------------------------------------
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo
;p7=indx
;p8=rev send
;p9=carrier/modulator frequency ratio
;p10=atk (0 => p3/2)
;p11=release (if p11==0 then p3/2)
;p12=index atk (if p12==0, then p3/2)
;p13=index release(if p13==0 then p3/2)
instr 3
iatk = p3/2-.001
idec = iatk
idxatk = iatk
idxdec = iatk
indx = p7
ifrq = cpspch(p5)
iamp = ampdb(p4)
;---- amplitude envelope calculation
if (p10+p11) == 0 igoto envok
if p10!=0 && p11!=0 igoto calcall
if p10==0 && p11!=0 igoto calcatk
if p10!=0 && p11==0 igoto calcdec
calcall:
iatk = p10
idec = p11
goto envok
calcatk:
idec = p11
iatk = p3-p11-.001
goto envok
calcdec:
iatk = p10
idec = p3-p10-.001
envok:
;--------- FM index envelope calculation
if (p12+p13) == 0 igoto ienvok
if p12!=0 && p13!=0 igoto icalcall
if p12==0 && p13!=0 igoto icalcatk
if p13!=0 && p13==0 igoto icalcdec
icalcall:
idxatk = p12
idxdec = p13
goto ienvok
icalcatk:
idxdec = p13
idxatk = p3-p13-.001
goto ienvok
icalcdec:
idxatk = p12
idxdec = p3-p12-.001
ienvok:
;--------- amplitude envelope
kenv linseg 0, iatk, iamp, p3 - iatk - idec, iamp, idec, 0
;--------- FM index envelope
kindx linseg 0, idxatk, indx, p3-idxatk-idxdec, indx, idxdec, 0
;--------- sound generation
a1 foscil kenv, ifrq, 1, p9, kindx, 1
ga1 = ga1 + a1 * p8
outs a1 * (1-p6), a1 * p6
endin
;-------------------------------------------------------
; INSTRUMENT 4 FLOU
;-------------------------------------------------------
instr 4
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo
;p7=atak (0 => atk=1)
;p8=rev send
;p9=ik frequency multiplication factor (in the style of McAdams)
iatak = p7
iatak = (p7 = 0. ? 1 : p7 )
ifrq = cpspch(p5)
iamp = ampdb(p4)
ik = (p9==0 ? 1:p9)
kenv1 linseg 0,.89*iatak,.4,p3-.89*iatak,0
kenv2 linseg 0,.83*iatak,.5,p3-.83*iatak,0
kenv3 linseg 0,.80*iatak,.4,p3-.80*iatak,0
kenv4 linseg 0,.95*iatak,.6,p3-.95*iatak,0
kenv5 linseg 0,.90*iatak,.5,p3-.90*iatak,0
kenv6 linseg 0,.70*iatak,.3,p3-.70*iatak,0
kenv7 linseg 0,.65*iatak,.4,p3-.65*iatak,0
krnd1 randi .2,3,-1
krnd2 randi .2,4,-1
krnd3 randi .2,2,-1
krnd4 randi .2,5,-1
krnd5 randi .2,6,-1
krnd6 randi .2,1,-1
krnd7 randi .2,2,-1
a1 oscil kenv1*krnd1,ifrq,1
a2 oscil kenv2*krnd2,ifrq*1.23*ik,1
a3 oscil kenv3*krnd3,ifrq*1.17*ik,1
a4 oscil kenv4*krnd4,ifrq*1.67*ik,1
a5 oscil kenv5*krnd5,ifrq*2.11*ik,1
a6 oscil kenv6*krnd6,ifrq*2.24*ik,1
a7 oscil kenv7*krnd7,ifrq*2.36*ik,1
atot = (a1+a2+a3+a4+a5+a7+a7)*iamp
ga1 = ga1+atot*p8
outs atot*(1-p6),atot*p6
endin
;-------------------------------------------------------
; INSTRUMENT 5 BASS-DRUM LIKE NOISE
;-------------------------------------------------------
instr 5
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
ifrq = cpspch(p5)
iamp = ampdb(p4)
a1 rand iamp
kenv linseg 1,.2,.3,p3-.2,0
a1 = a1*kenv
kfilt linseg ifrq*2,p3,ifrq/2
a2 reson a1,kfilt,kfilt/2,2
ga1 = ga1 + a2*p8
outs a2*(1-p6),a2*p6
endin
;-------------------------------------------------------
; INSTRUMENT 6 “WANDERING” TUBA
;-------------------------------------------------------
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=attack type; 0=short, 1=long
;p8=random init seed
instr 6
ioct = octpch(p5)
iamp = ampdb(p4)
inindx init 0
icntindx init 7
icar init 1
imod init 1
krnd randi .01,2,p8/2
kfrq = cpsoct(ioct + krnd)
;------------- amplitude envelope generation
if p7>0 kgoto long
kamp linseg 0,p3/8,iamp,p3*6/8-.01,iamp,p3/8,0.,.01,0
kgoto longdone
long:
kamp linseg 0,p3/4,iamp,p3*3/4-.01,0,.01,0
longdone:
;------------ FM sound generation
kind linseg inindx,p3/2,icntindx,p3/2,inindx
a5 foscili kamp,kfrq,icar,imod,kind,1
ga5 atone a5, 20
; ----------- stereo control
if p6 >= 0 goto stereo
kst randi .5,3,p8
kst = kst + .5
kgoto donest
stereo:
kst = p6
donest:
; ----------------------------
outs ga5 * (1-kst)*2, ga5 * kst*2
endin
;-------------------------------------------------------
; INSTRUMENT 7 LOSCIL
;-------------------------------------------------------
instr 7
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p8=rev.send
;p9=original pitch (oppc)
;p10=function no.
;p11=attack time
;p12=release time
ifrq = cpspch(p5)
ifrq0 = cpspch(p9)
iamp = ampdb(p4)
kenv linseg 0,p11,iamp,p3-p11-p12,iamp,p12,0
a1 loscil kenv,ifrq,p10,ifrq0
ga1 = ga1+a1
outs a1*(1-p6),a1*p6
endin
;-------------------------------------------------------
; INSTRUMENT 8 RANDOM NOISE (SELECTABLE BANDWIDTH)
;-------------------------------------------------------
instr 8
;p4=amplitude in dB
;p5=filter starting center pitch (oppc)
;p6=stereo position
;p7=bandwidth
;p8=rev. send
;p9= filter ending center pitch(if p9==0, then no frequency ; variation)
iinoct = octpch(p5)
ifinoct = octpch(p9)
iamp = ampdb(p4)
if p9!= 0 goto finfrq
ifinoct = iinoct
finfrq:
kfrq line iinoct,p3,ifinoct
kfrq = cpsoct(kfrq)
kenv linseg 0,p3/2,1,p3/2,0
a1 rand iamp*kenv
a2 butterbp a1,kfrq,p7
ga1 = ga1 + a2 * p8
outs a2 *(1-p6),a2*p6
endin
;-------------------------------------------------------
; INSTRUMENT 9 GROUPS OF FILTERED PULSES
;-------------------------------------------------------
instr 9
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=bandwidth (cps)
;p8=rev.send
;p9=starting pulse duration
;p10=ikdur (<1)
;p11=ikfrq
idur = p9
ikdur = p10
ifrq = cpspch(p5)
iamp = ampdb(p4)
ibw = p7
ikfrq = p11
kenv line 1,p3,0
fromhere:
a1 linseg iamp,.0025,0,p3-.0025,0
a2 butterbp a1,ifrq,ibw
a2 = a2*kenv
ga1 = ga1+a2*p8
timout 0,idur,contin
idur = idur*ikdur
ifrq = ifrq*ikfrq
reinit fromhere
contin:
outs a2*(1-p6),a2*p6
endin
;-------------------------------------------------------
; INSTRUMENT 10 SAMPLED BASS DRUM
;-------------------------------------------------------
instr 10
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=ending pitch (oppc)(if p7==0 , no glissando)
;p8=rev.send
ifrq = cpspch(p5)
iamp = ampdb(p4)
kenv linseg iamp,p3,0
if p7==0 goto nogliss
kfrq expseg ifrq,p3,cpspch(p7)
goto ok
nogliss:
kfrq linseg ifrq,p3,ifrq
ok:
kdump kfrq,"displayz.bin",6,.001
a1 loscil kenv,kfrq,10,cpspch(7)
ga1 = ga1 + a1 * p8
outs a1*(1-p6),a1*p6
endin
;-------------------------------------------------------
; INSTRUMENT 11 RANDOM INDEX FM
;-------------------------------------------------------
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo
;p7=index
;p8=rev.send
;p9=carrier/modulator ratio
;p10=amplitude envelope attack time(0 => p3/2)
;p11=amplitude envelope release time (0 => p3/2)
;p12=index envelope attack time (0 => p3/2)
;p13=index envelope release time (0 => p3/2)
instr 11
iatk = p3/2-.001
idec = iatk
idxatk = iatk
idxdec = iatk
indx = p7
ifrq = cpspch(p5)
iamp = ampdb(p4)
if (p10+p11) == 0 igoto envok
if p10!=0 && p11!=0 igoto calcall
if p10==0 && p11!=0 igoto calcatk
if p10!=0 && p11==0 igoto calcdec
calcall:
iatk = p10
idec = p11
goto envok
calcatk:
idec = p11
iatk = p3-p11-.001
goto envok
calcdec:
iatk = p10
idec = p3-p10-.001
envok:
if (p12+p13) == 0 igoto ienvok
if p12!=0 && p13!=0 igoto icalcall
if p12==0 && p13!=0 igoto icalcatk
if p13!=0 && p13==0 igoto icalcdec
icalcall:
idxatk = p12
idxdec = p13
goto ienvok
icalcatk:
idxdec = p13
idxatk = p3-p13-.001
goto ienvok
icalcdec:
idxatk = p12
idxdec = p3-p12-.001
ienvok:
kenv linseg 0, iatk, iamp, p3 - iatk - idec, iamp, idec, 0
kindx linseg 0, idxatk, indx, p3-idxatk-idxdec, indx, idxdec, 0
kindrnd randi .5,2,frac(p5) ;variazione random indice
kindrnd = .5+abs(kindrnd)
a1 foscil kenv, ifrq, 1, p9, kindx*kindrnd, 1
ga1 = ga1 + a1 * p8
outs a1 * (1-p6), a1 * p6
endin
;-------------------------------------------------------
; INSTRUMENT 12 PLUCK (KARPLUS & STRONG)
;-------------------------------------------------------
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=function no. (0=random)
;p8=icps (max table length)
;p9=rev. send
instr 12
ifrq = cpspch(p5)
iamp = ampdb(p4)
icps = (p8==0? ifrq:p8)
kenv linseg iamp,p3,0
; kamp,kcps,icps,ifn,meth
a1 pluck kenv,ifrq,icps,p7, 1
ga1 = ga1 + a1
outs a1*(1-p6),a1*p6
endin
;-------------------------------------------------------
; INSTRUMENT 13 FM (FOR “DEJO MIS VIEJOS LIBROS...”)
;-------------------------------------------------------
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p7=FM index
;p8=rev send
;p9=carrier/modulator frequency ratio
;p10=tied to note(0=no, 1=previous, 2=next, 3=both)
;p11=next pitch (if tied to next note)
instr 13
iatk = p3/3
icm = p9
indx = p7
ifrq = cpspch(p5)
infrq = cpspch(p11)
igliss = .02 ;gliss time
iamp = ampdb(p4)
if p10==0 goto notied
if p10==1 goto tiedprev
if p10==2 goto tiednext
if p10==3 goto tiedboth
tiedprev:
kenv linseg iamp*.5, iatk, iamp,p3 - 2*iatk , iamp*.5, iatk,0
kindx linseg 1, iatk,indx, p3- iatk, 0
kfrq line ifrq,p3,ifrq
iphs = -1
goto tiedok
tiednext:
kenv linseg 0, iatk, iamp,p3 - 2*iatk , iamp*.5, iatk,iamp*.5
kindx linseg 0, iatk,indx, p3- iatk, 1
kfrq linseg ifrq,p3-igliss,ifrq,igliss,infrq
iphs = -1
goto tiedok
tiedboth:
kenv linseg iamp*.5, iatk, iamp, p3 - 2*iatk , iamp*.5, iatk,iamp*.5
kindx linseg 1, iatk,indx, p3- iatk, 1
kfrq linseg ifrq,p3-igliss,ifrq,igliss,infrq
iphs = -1
goto tiedok
notied:
kenv linseg 0, iatk, iamp, p3 - 2*iatk , iamp*.5, iatk,0
kindx linseg 0, iatk,indx, p3- iatk, 0
kfrq line ifrq,p3,ifrq
iphs = 0
tiedok:
kindx1 randi .4,1,-1
kindx1 = 1+kindx1+.2
a1 foscil kenv, kfrq, 1, icm, kindx*kindx1, 1,iphs
ga1 = ga1 + a1
outs a1 * (1-p6),a1*p6
endin
;-------------------------------------------------------
; INSTRUMENT 15 ADSYN 1
;-------------------------------------------------------
instr 15
;p4=amplitude in dB
;p5=pitch (oppc)
;p6=stereo position
;p8=rev send
ifrq = cpspch(p5)
ifrq0 = cpspch(8.09)
iamp = ampdb(p4)/32000
kenv linseg iamp,p3*.8,iamp,p3*.2,0
kspeed linseg 1,.04,1,.001,1.94/p3,p3-.04,1.94/p3
a1 adsyn kenv,ifrq/ifrq0,kspeed,"tr1b.het"
a2 butterlp a1,4000
a3 butterlp a2,4000
a4 butterlp a3,4000
ga1 = ga1+a4*p8
outs a4*(1-p6),a4*p6
endin
;-------------------------------------------------------
; INSTRUMENT 96 VARIABLE DELAY
;-------------------------------------------------------
instr 96
;p4=max delay time
;p5=feedback
;p6=stereo position
a1 init 0
itime = p4
ifeed = p5
;kdel1 oscil itime*.45,.332,1
kdel1 randi itime*.45,3.27,-1
kdel2 randi itime*.45,4.11,-1
kdel3 randi itime*.45,4.93,-1
kdel1 = itime+kdel1
kdel2 = itime+kdel2
kdel3 = itime+kdel3
a0 delayr itime*2
delayw ga2+a1*p5
a1 deltapi kdel1
a2 deltapi kdel1
a3 deltapi kdel3
aout = (a1+a2+a3)/3
outs aout*(1-p6),aout*p6
ga2 = 0
endin
;-------------------------------------------------------
; INSTRUMENT 97 DELAY 1
;-------------------------------------------------------
instr 97
;p4=delay time
;p5=feedback
;p6=stereo position
itime = p4
ifeed = p5
a1 delayr p4
delayw ga2+a1*p5
outs a1*(1-p6),a1*p6
ga2 = 0
endin
;-------------------------------------------------------
; INSTRUMENT 98 DELAY 2
;-------------------------------------------------------
instr 98
;p4=delay time
;p5=feedback
;p6=stereo position
itime = p4
ifeed = p5
a1 delayr p4
delayw ga3+a1*p5
outs a1*(1-p6),a1*p6
ga3 = 0
endin
;-------------------------------------------------------
; INSTRUMENT 99 REVERB
;-------------------------------------------------------
instr 99
imix = p4
krvtim = p5
a1 alpass ga1*p4,krvtim,.04996
a2 alpass a1*.72, krvtim, .05465
a3 alpass a2*.961, krvtim, .02418
al alpass a3*.649, krvtim, .01785
ar alpass a3*.646, krvtim, .01795
outs al,ar
ga1 = 0
endin
