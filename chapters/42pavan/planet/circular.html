Circular Sounds

3d effect in a stereophonic environment

Input: a mono soundfile

Output: a stereo soundfile

Orchestra

;==============================================================================;


;                                                                              ;
;                                Circular sounds                               ;
;                                                                              ;
;                    3D effect in a stereophonic environment                   ;
;                                                                              ;
;                                   Orchestra                                  ;
;                                                                              ;
;                               Luca Pavan - 1997                              ;
;                                                                              ;
;==============================================================================;
sr = 44100
kr = 44100
ksmps  = 1
nchnls = 2

instr 1

   idur      = p3     ;duration
   iamp      = p4     ;amplitude (1 = original amplitude)
   iskiptime = p5     ;skiptime
   iattack   = p6     ;release
   irelease  = p7     ;decay
   ifcmin    = p8     ;minimal central frequency
   ifcmax    = p9     ;maximal central frequency
   kband     = p10    ;bandwidth
   kpan1     = p11    ;pan left channel
   kpan2     = p12    ;pan right channel
   icircles  = p13    ;number of sound rotations
                                      
   kamp		linen	iamp, iattack, idur, irelease
   kpan1        oscili  kpan1, (1/p3) * icircles, 6
   kpan2        oscili  kpan2, (1/p3) * icircles, 5
   asig         soundin "voice.wav", iskiptime
   arampsig  =  kamp * asig

   kcentro      oscili  ifcmax, (1/p3) * icircles, 7
   autfilt1     butterbp arampsig * kpan1, ifcmin + kcentro, kband
   autfilt2     butterbp arampsig * kpan2, ifcmin + kcentro, kband
     
     outs autfilt1, autfilt2
       endin

Score

f1 0  262144 -1 "voice.wav" 0 0 0
f5 0 512 7 1 256 0 256 1
f6 0 512 7 0 256 1 256 0
f7 0 1024 7 0 256 1 256 0 512 0

i1    0     5    2   0	 .03  .1   200     5000     5000   1   1    3   
e
