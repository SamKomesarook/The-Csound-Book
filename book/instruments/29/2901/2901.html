  instr 2901 ; pulse wave with modulatable pulse-width
iatk = .2
idec = .1
isus = .8
irel = .2
istdy = 1-(iatk+idec+irel) ; length of sustain level
alfo oscili 0.2, p6, 3 ; p6=lfo rate

apulse table asaw-apw, 1, 1, 0.5
kenv linseg 0, p3*iatk, 1, p3*idec, isus, p3*istdy, isus, p3*irel, 0
  out p4*apulse*kenv
